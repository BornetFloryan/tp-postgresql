# Sauvegarde logique

pg_dump -U postgres -d multi_lang --no-owner --no-acl -f multi_lang.sql

# Création de la base de restauration

createdb -U postgres BUT3_backup

# Restauration complète

psql -U postgres -d BUT3_backup -f multi_lang.sql

# Dump custom

pg_dump -U postgres -d multi_lang -Fc --no-owner --no-acl -f multi_lang.dump

# Dump custom limité à la table étudiants avec ses objets dépendants

pg_dump -U postgres -d multi_lang -Fc --no-owner --no-acl -t etudiants -f multi_lang_etudiants.dump

# Base vide pour la restauration partielle

createdb -U postgres BUT3_etudiants_only

# Restauration complète de la table étudiants, sa séquence et ses contraintes

pg_restore -U postgres -d BUT3_etudiants_only multi_lang_etudiants.dump
