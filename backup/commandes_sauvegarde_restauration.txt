# Sauvegarde logique

pg_dump -U postgres -d multi_lang -f multi_lang.sql

# Création de la base de restauration

createdb -U postgres BUT3_backup

# Restauration complète

psql -U postgres -d BUT3_backup -f multi_lang.sql

# Dump limité à la table étudiants

pg_dump -U postgres -d multi_lang -Fc -t etudiants -f multi_lang_etudiants.dump

# Base vide pour la restauration ciblée

createdb -U postgres BUT3_etudiants_only

# Restauration de la table étudiants

pg_restore -U postgres -d BUT3_etudiants_only multi_lang_etudiants.dump
