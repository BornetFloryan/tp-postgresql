pg_dump -U postgres -d multi_lang -f multi_lang.sql

createdb -U postgres BUT3_backup

psql -U postgres -d BUT3_backup -f multi_lang.sql

createdb -U postgres BUT3_etudiants_only

pg_restore -U postgres -d BUT3_etudiants_only -t etudiants multi_lang.dump
